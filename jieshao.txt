const http = require('http')
http.createServer((req,res)=>{
    const {url} = req
    if(url === '/'){ //如果没有main.js的话，添加main.js
        res.end(`<html></html>`)
    }
}).listen(3000,()=>{
    console.log('Http Cache Test Run at '+ 3000)
})
